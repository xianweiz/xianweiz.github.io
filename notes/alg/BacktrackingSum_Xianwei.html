0. Backtracking

The usual scenario is that you are faced with a number of options, and you must choose one of these. After you make your choice you will get a new set of options; just what set of options you get depends on what choice you made. This procedure is repeated over and over until you reach a final state. If you made a good sequence of choices, your final state is a goal state; if you didn't, it isn't.

0.1 Find a path to success (is there a solution?)
bool solve(Node n) {
  if n is a leaf node {
    if the leaf is a goal node, return true
    else return false
  }
  for each child c of n {
    if solve(c) succeeds, return true
  }
  return false
}
0.2 Find all paths to success (list or count all solutions)
void solve(Node n) {
  if n is a leaf node {
    if the leaf is goal node, count++
    return
  }
  for each child c of n
  	solve(c)
}
0.3 Find the best path to success
void solve(Node n) {
  if n is a leaf node {
    if the leaf is a goal node, update the best result
    return
  }
  for each child c of n
  	solve(c)
}

LC351: Android unlock patterns

Given an Android 3x3 key lock screen and two integers m and n, where 1 ≤ m ≤ n ≤ 9, count the total number of unlock patterns of the Android lock screen, which consist of minimum of m keys and maximum n keys.

 
int helper(int num, int len, int count, int m, int n, vector<bool>& visited, 
           vector<vector<int>>& jumps) {
  if(len >= m) ++ count;
  ++ len;
  if(len > n) return count;
        
  visited[num] = true;  
  for(int i=1; i<10; ++i) {
    int next = i, jump = jumps[num][next];
    //if 'next' has NOT been visited, and no jump or the jump has been visited
    if(!visited[next] && (jump==0 || visited[jump])) {
      count = helper(next, len, count, m, n, visited, jumps);
    }
  }//--for 
  visited[num] = false;
  return count;
}
int numberOfPatterns(int m, int n) {
  vector<vector<int>> jumps(10, vector<int>(10, 0));
  //needs init
  vector<bool> visited(10, false);
  int res = 0;
  res += helper(1, 1, 0, m, n, visited, jumps) * 4;   //1,3,7,9
  res += helper(2, 1, 0, m, n, visited, jumps) * 4;   //2,4,6,8
  res += helper(5, 1, 0, m, n, visited, jumps);       //5
  return res;
}

LC39: Combination sum

Given a set of candidate numbers (C) (without duplicates) and a target number (T), find all unique combinations in C where the candidate numbers sums to T.

The same repeated number may be chosen from C unlimited number of times.

without duplicates: the orig is are distinct --> just move forward (int i=pos)
unlimited #times: each number can be repeatedly used, and hence use 'i' to recursive
 
vector<vector<int>> rst;
void helper(vector<int>& nums, vector<int>& path, int target, int pos) {
  if(target <= 0) {if(target==0) rst.push_back(path); return;}
  for(int i=pos; i<nums.size(); ++i) {      //'pos' to advoid duplicate combination
    int num = nums[i];
    if(num > target) continue;
    path.push_back(num);
    helper(nums, path, target-num, i);      //'i' to repeatedly use num
    path.pop_back();
  }//--for
}
vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
  vector<int> path;
  helper(candidates, path, target, 0);
  return rst;
}
LC40: Combination sum II

Given a collection of candidate numbers (C) and a target number (T), find all unique combinations in C where the candidate numbers sums to T.

Each number in C may only be used once in the combination.

the orig set is NOT distinct --> move forward (int i=pos), and skip same numbers ([i]==[i-1]) aft sort
only used once: number cannot be repeatedly used, and hence use 'i+1' to recursive
 
vector<vector<int>> rst;
void helper(vector<int>& nums, vector<int>& path, int target, int pos) {
  if(target <= 0) {if(target==0) rst.push_back(path); return;}
  for(int i=pos; i<nums.size(); ++i) {      //'pos' to advoid duplicate combination
    int num = nums[i];
    if(num > target) continue;
    if(i>pos && nums[i-1]==nums[i]) continue;//skip same number
    path.push_back(num);
    helper(nums, path, target-num, i+1);    //'i+1' to avoid repeatedly use num
    path.pop_back();
  }//--for
}
vector<vector<int>> combinationSum(vector<int>& candidates, int target) {
  vector<int> path;
  std::sort(candidates.begin(), candidates.end());  //sort
  helper(candidates, path, target, 0);
  return rst;
}

LC377: Combination sum IV

Given an integer array with all positive numbers and no duplicates, find the number of possible combinations that add up to a positive integer target.

no duplicates: set numbers are distinct
# of psbl combinations: no need to list all
 
int combinationSum4(vector<int>& nums, int target) {
  std::vector<int> dp(target+1, 0);           //combs to get 'i'
  dp[0] = 1;
  for(int k=1; k<=target; ++k)
    for(auto a : nums)
      if(k>=a) dp[k] += dp[k-a];
  return dp[target];
}

bool canWin(string s) {
  vector<string> psblMovs;

bool helper(int i, int j, vector<vector<char>>& board) {
  if(i == 9) return true;                   //all have been filled
  if(j == 9) return helper(i+1, 0, board);    //reach right boundary, start next row
        
  if(board[i][j] != '.') return helper(i, j+1, board); //alreay has digit, consider next col
        
  for(char c='1'; c<='9'; ++c) {
    if(!isValid(c, i, j, board)) continue;
    board[i][j] = c;
    if(helper(i, j+1, board)) return true;
    board[i][j] = '.';
  }//--for
  return false;
}

  helper(0, 0, board);

LC320: Generalized abreviation

Write a function to generate the generalized abbreviations of a word.

for each start position i, gradually replace 1 to n-i chars
generally, start position should be one behind last end position to avoid consecutive seperated digits (exception is '0')
helper(string curr, string word, int start, vector<int> rst)
  ==> helper("", word, 0, rst)
  ==> word : curr="" --> start=1/2/3/4, w+start+word.substr(start)
 
void helper(const string word, string curr, int start,  vector<string>& rst) {
  rst.push_back(curr + word.substr(start));
  if(start == word.length()) return;
        
  int i = (start==0) ? 0 : start+1;
  for(; i<word.length(); ++i) {             //try all psbl start pos, i.e., replace after pos
    string prefix = curr + word.substr(start, i-start);
    for(int j=1; j<=word.length()-i; ++j) { //try all psbl replacement width, [1..len-i]
      helper(word, prefix+std::to_string(j), i+j, rst);
    }//--for
  }//--for
}
vector<string> generateAbbreviations(string word) {
  vector<string> rst;        
  helper(word, "", 0, rst);
  return rst;
}

bool validWordAbbreviation(string word, string abbr) {
  int j = 0, i = 0;
  for(i=0; i<abbr.length();){
    if(!std::isdigit(abbr[i])) {               //not a digit
      if(word[j] != abbr[i]) return false;
      else { ++i; ++j; }
    } else {
      if(abbr[i] == '0') return false;    //cannot start with '0'
      int start = i;
      while(std::isdigit(abbr[i])) {
        ++ i;
      }
      int cnt = std::stoi(abbr.substr(start, i-start));
      j += cnt;
    }
  }//--for
  return i==abbr.size() && j==word.size();    //CANNOT directly return true
}

//Backtracking + Trie
struct TrieNode {
  vector<string> prefixWords;
  vector<TrieNode*> children;
  TrieNode() { children.resize(26, nullptr);}
};
TrieNode* root = new TrieNode();
void buildTrie(vector<string>& words) {
  for(auto word : words) {
    TrieNode* node = root;
    for(char c : word) {
      node->prefixWords.push_back(word);
      if(node->children[c-'a'] == nullptr) {          //not exist
        node->children[c-'a'] = new TrieNode(); //create the node
      }
      node = node->children[c-'a'];                   //move downward
    }//--for
  }//--for
}
void searchTrie(string prefix, vector<string>& rst) {
  TrieNode* node = root;
  for(int i=0; i<prefix.size() && node!=nullptr; ++i){
    node = node->children[prefix[i]-'a'];
  }//--for
  if(node == nullptr) return;
  for(string wd : node->prefixWords) {
    rst.push_back(wd);
  }
}
string getPrefix(const vector<string>& path) {
  int n = path.size();
  string prefix = "";
  for(int i=0; i<n; ++i) {
    prefix += path[i][n];
  }
  return prefix;
}
void helper(vector<string>& words, const int len, vector<string>& path, vector<vector<string>>& rst) {
  if(path.size() == len) {
    rst.push_back(path); return;
  }
        
  string prefix = getPrefix(path);
  vector<string> cands; searchTrie(prefix, cands);
  for(int i=0; i<cands.size(); ++i) {
    string word = cands[i];
    path.push_back(word);
    helper(words, len, path, rst);
    path.pop_back();
  }//--for
}

vector<string> readBinaryWatch(int num) {
    vector<string> rs;
    for (int h = 0; h < 12; h++)
    for (int m = 0; m < 60; m++)
        if (bitset<10>(h << 6 | m).count() == num)
            rs.emplace_back(to_string(h) + (m < 10 ? ":0" : ":") + to_string(m));
    return rs;
}
