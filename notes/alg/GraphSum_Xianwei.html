unordered_map<GraphNode*, GraphNode*> mp;
UndirectedGraphNode* cloneGraph2(UndirectedGraphNode* node){
  if(node == nullptr) return node;
  if(mp.find(node) != mp.end()) return mp[node];                //has been duplicated
  UndirectedGraphNode* copy = new UndirectedGraphNode(node->label);
  mp[node] = copy;
  for(auto nghbr : node->neighbors){                            //copy neighbors
    mp[node]->neighbors.push_back(cloneGraph(nghbr));
  }//--for
  return mp[node];
}
 
//BFS
UndirectedGraphNode* cloneGraph(UndirectedGraphNode* node){
  if(node == nullptr) return node;
  UndirectedGraphNode* copy = new UndirectedGraphNode(node->label);
  mp[node] = copy;
  queue<UndirectedGraphNode*> qu;
  qu.push(node);
        
  while(!qu.empty()) {
    UndirectedGraphNode* tmpNode = qu.front(); qu.pop();
    for(auto nghbr : tmpNode->neighbors) {              //scan neighbors
      if(mp.find(nghbr) == mp.end()) {
        mp[nghbr] = new UndirectedGraphNode(nghbr->label);
        qu.push(nghbr);
      }
      mp[tmpNode]->neighbors.push_back(mp[nghbr]);
    }//--for
  }//--while
  return mp[node];
}

LC310: Minimum height trees

For a undirected graph with tree characteristics, we can choose any node as the root. The result graph is then a rooted tree. Among all possible rooted trees, those with minimum height are called minimum height trees (MHTs). Given such a graph, write a function to find all the MHTs and return a list of their root labels.

First, put all leaf nodes (degree=1) into the queue, then they are removed from tree, and then push new leaves into the queue; repeat until only 1 or 2 nodes are left.
 
//O(N): touch each node once
vector<int> findMinHeightTrees(int n, vector<pair<int, int>>& edges) {
  if(n == 1) return {0};
  
  vector<int> degrees(n, 0);
  vector<vector<int>> adjList = makeGraph(n, edges, degrees);
        
  queue<int> qu;
  for(int i=0; i<n; ++i)
    if(degrees[i] == 1) qu.push(i);
        
  int totalV = n;
  while(totalV > 2) {
    int size = qu.size();
    totalV -= size;
    for(int i=0; i<size; ++i) {
      int v = qu.front(); qu.pop();
      //-- totalV;
      for(auto nghbr : adjList[v]){
        -- degrees[nghbr];
        if(degrees[nghbr] == 1) qu.push(nghbr);
      }
    }
  }//--while
        
  vector<int> rst;
  while(!qu.empty())
    rst.push_back(qu.front()); qu.pop();
  return rst;
}

Given n nodes labeled from 0 to n - 1 and a list of undirected edges (each edge is a pair of nodes), write a function to check whether these edges make up a valid tree.

a tree is an undirected graph in which any two vertices are connected by exactly one path.

std::vector<std::vector<int>> makeGraph(int n, vector<pair<int, int>>& edges) {
  std::vector<std::vector<int>> adjList(n, std::vector<int>());
  for(auto edge : edges) {
    int src = edge.first;
    int dst = edge.second;
    adjList[src].push_back(dst);
    adjList[dst].push_back(src);
  }//--for
  return adjList;
}
bool validTree(int n, vector<pair<int, int>>& edges) {
  if(n == 0) return true;
  if(edges.size()!=n-1) return false;             //cond1: MUST be n-1 edges
                                                  //cond2: MUST be connected
  std::vector<std::vector<int>> adjList = makeGraph(n, edges);
  std::vector<bool> visited(n, false);
  std::queue<int> qu;
  qu.push(0); visited[0] = true;
        
  int cnt = 0;
  while(!qu.empty()) {
    int node = qu.front(); qu.pop();
    ++ cnt;
    for(auto nghbr : adjList[node]) {
      if(!visited[nghbr]) {
        qu.push(nghbr); visited[nghbr] = true;
      }
    }//--for
  }//--while
  return cnt==n;
}

  1. Only one letter can be changed at a time.
  2. Each transformed word must exist in the word list. Note that beginWord is not a transformed word.
先把起点加到队列中, 然后每次将字典中与队首距离为1的字符串加进队列, 直到最后出队列的是终点字符串, 为确保终点字符串存在, 我们可以先在字典中加进去终点字符串.
 
//BFS to find shortest distance from 'beginWord' to 'endWord'
//queue: pop the front, and add its 1HD words, i.e., neighbors
//unused word set: initiall all, and keep erasing when added as neighbors to avoid access again
int ladderLength(string beginWord, string endWord, vector<string>& wordList) {
  unordered_set<string> words;
  for(auto word : wordList)
    words.insert(word);
  queue<string> qu;
  qu.push(beginWord);
  words.erase(beginWord);
  
  int minSteps = 2;
  while(!qu.empty()){
    int size = qu.size();
    for(int i=0; i<size; ++i){
        string word = qu.front(); qu.pop;
        for(auto nghbr : get1HDwords(word)){
            if(nghbr == endWord) return minSteps;
            qu.push(nghbr);
        }
    }
    ++ minSteps;
  }//--while
  return 0;
}

1.3 Topological Sort

  • BFS based

    start from with vertices with 0 incoming edge，insert them into list S，at the same time we remove all their outgoing edges，after that find new vertices with 0 incoming edges and go on.

  • DFS based

    loop through each node of the graph in an arbitrary order，initiating a depth-first search that terminates when it hits any node that has already been visited since the beginning of the topological sort or the node has no outgoing edges (i.e. a leaf node).

LC207: Course schedule

There are a total of n courses you have to take, labeled from 0 to n - 1.

Some courses may have prerequisites, for example to take course 0 you have to first take course 1, which is expressed as a pair: [0,1]

Given the total number of courses and a list of prerequisite pairs, is it possible for you to finish all courses?

This problem is equivalent to finding if a cycle exists in a directed graph. If a cycle exists, no topological ordering exists and therefore it will be impossible to take all courses.
 
//BFS: time-O(E+V), space-O(V)
vector<vector<int>> makeGraph(int n, vector<pair<int, int>>& edges){
  vector<vector<int>> adjList(n, vector<int>());
  for(auto edge : edges) {
    int dst = edge.first;
    int src = edge.second;
    adjList[src].push_back(dst);
  }//--for
  return adjList;
}
void getDegrees(vector<pair<int, int>>& edges, vector<int>& indegrees){
  for(auto edge : edges) {
    int dst = edge.first;
    ++ indegrees[dst];    
  }//--for
}
bool canFinish(int numCourses, vector<pair<int, int>>& prerequisites) {
  vector<vector<int>> adjList = makeGraph(numCourses, prerequisites);   //O(E)
  vector<int> indegrees(numCourses, 0);
  getDegrees(prerequisites, indegrees);                               //O(E)
  
  queue<int> qu;
  vector<bool> visited(numCourses, false);
  for(int i=0; i<indegrees.size(); ++i)                 //first push 0-degree vertices
    if(indegrees[i] == 0) { qu.push(i); visited[i]=true; }
  
  vector<int> rst;
  while(!qu.empty()) {                                  //O(V+E)
    int course = qu.front(); qu.pop();
    rst.push_back(course);
    for(auto nghbr : adjList[course]){
      -- indegrees[nghbr];
      if(!visited[nghbr] && indegrees[nghbr]==0){
        qu.push(nghbr); visited[nghbr]=true;
      }
    }//--for
  }//--while
  return rst.size()==indegrees.size();
}
LC210: Course schedule II

Given the total number of courses and a list of prerequisite pairs, return the ordering of courses you should take to finish all courses. There may be multiple correct orders, you just need to return one of them. If it is impossible to finish all courses, return an empty array.

For DFS, it will first visit a node, then one neighbor of it, then one neighbor of this neighbor... and so on. If it meets a node which was visited in the current process of DFS visit, a cycle is detected and we will return false. Otherwise it will start from another unvisited node and repeat this process till all the nodes have been visited. Note that you should make two records: one is to record all the visited nodes and the other is to record the visited nodes in the current DFS visit.

The code is as follows. We use a vector<bool> visited to record all the visited nodes and another vector<bool> onpath to record the visited nodes of the current DFS visit. Once the current visit is finished, we reset the onpath value of the starting node to false.
 
bool DFS(int node, vector<vector<int>>& adjList, vector<int>& rst, 
         vector<bool>& visited, vector<bool>& onpath) {
  if(onpath[v]) return false;       //on the same DFS path
  if(visited[v]) return true;       //visited, no need to go further
  visited[node] = onpath[node] = true;
  for(auto nghbr : adjList[node]) {
    if(onpath[nghbr] || !DFS(nghbr, adjList, rst, visited, onpath))
      return false;
  }//--for
  rst.push_back(node);
  onpath[node] = false;
  return true;
}
vector<int> findOrder(int numCourses, vector<pair<int, int>>& prerequisites) {
  vector<vector<int>> adjList = makeGraph(numCourses, prerequisites);

LC444: Sequence reconstruction

Check whether the original sequence org can be uniquely reconstructed from the sequences in seqs. 

Reverse of classical topological sort:
1) build graph letting src->dst
2) stat out degrees, instead of in degrees
3) apply the provided org seq to see whether we can uniquely repeat
 
bool sequenceReconstruction(vector<int>& org, vector<vector<int>>& seqs) {
  if(seqs.empty()) return false;
  //build graph
  int n = org.size(), count=0;
  unordered_map<int, unordered_set<int>> adjList;
  vector<int> degrees(n+1, 0);
  unordered_set<int> Vs;
  for(auto seq : seqs) {
    for(int i=0; i<seq.size(); ++i){
      int dst = seq[i];
      Vs.insert(dst);
      if(dst>n || dst<0) return false;
      if(i>0 && !adjList[dst].count(seq[i-1])) {              //handle duplicate edges
        int src = seq[i-1]; Vs.insert(src);
        adjList[dst].insert(src);
        if(degrees[src]++ == 0) ++ count;
      }
    }
  }//--for
        
  for(auto v : org){                                              //handle extra nodes in 'org'
    if(Vs.find(v) == Vs.end()) return false;
  }
  
  if(count != n-1) return false;
        
  for(int i=n-1; i>=0; --i){
    if(degrees[org[i]] > 0) return false;
    for(auto nghbr : adjList[org[i]]){
      --degrees[nghbr];
      if(degrees[nghbr]==0 && nghbr!=org[i-1]) return false;
    }
  }//--for
  return true;
}

void DFS(string airport, unordered_map<string, multiset<string>>& adjList, vector<string>& rst){
  while(adjList[airport].size() > 0) {      //neighbors
    string to = *adjList[airport].begin();

  }
  rst.push_back(airport);

vector<string> findItinerary(vector<pair<string, string>> tickets) {
  if(tickets.size() == 0) return {};
  //make adjList
  unordered_map<string, multiset<string>> adjList;
  for(auto ticket : tickets) {
    string from = ticket.first;
    string to = ticket.second;
    adjList[from].insert(to);
  }
  vector<string> rst;
  DFS("JFK", adjList, rst);
  //We are always appending the deepest node to the itinerary, 
  // so will need to reverse the itinerary in the end.
  std::reverse(rst.begin(), rst.end());
  
  return rst;
}

LC399: Evaluate division

Equations are given in the format A / B = k, where A and B are variables represented as strings, and k is a real number (floating point number). Given some queries, return the answers. If the answer does not exist, return -1.0.

 
double queryGraph(const std::string& up, const std::string& down,
                  std::unordered_map<std::string, std::unordered_map<std::string, double>>& graph,
                      std::unordered_set<std::string>& visited) {
  visited.insert(up); 
        
  if (graph[up].find(down) != graph[up].end()) {
    return graph[up][down];
  }
  for (auto ipair : graph[up]) {
    if (visited.find(ipair.first) == visited.end()) {
      double q = queryGraph(ipair.first, down, graph, visited);
      if (q != -1)
        return q * ipair.second;
    }
  }
  return -1;
}
vector<double> calcEquation(vector<pair<string, string>> equations, vector<double>& values, vector<pair<string, string>> queries) {
  std::unordered_map<std::string, std::unordered_map<std::string, double>> graph;
  
  std::vector<double> result;
  int n = values.size();
  for (int i = 0; i < n; i++) {
    graph[equations[i].first][equations[i].second] = values[i];
    graph[equations[i].second][equations[i].first] = 1 / values[i];
  }
  for (auto query : queries) {
    std::unordered_set<std::string> visited;
    double quotient = queryGraph(query.first, query.second, graph, visited);
    result.push_back(quotient);
  }
  return result;
}

3. Trie

A trie is a variant of -ary tree in which characters are stored at each node. Each path down the tree may represent a word. A node in a trie could have anywhere from 1 through ALPHABET_SIZE children.

Very commonly, a trie is used to store the entire (English) language for quick prefix lookups. While a hash table can quickly look up whether a string is a valid word, it cannot tell us if a string is a prefix of any valid words. A trie can do this quickly.

A trie can check if a string is a valid prefix in time, where K is the length of the string.

  }

    TrieNode* node = find(prefix);

  }

      const char& c = word[i];
      node = node->children[c-'a'];
    }//--for

  }
